	//创建一个对象
	var box = new Object();
	box.name="zhouitpro";  //属性创建
	box.age = 19;
	
	//创建对象2,可以去掉new 也是可以的
	var box = Object();
	
	//创建对象3
	var box = {};
	
	box.name = "zhouitpro"
	
	//对象方式输出
	alert(box.name);
	
	//数组方式输出
	alert(box['name']);
	
	//创建对象方法
	box.run = function(){
		return 'run';
	}
	
	//删除属性
	delete box.name;
	
	//判断属性是否存在
	if(box.name != undefined){
		alert(box.name);
	}
