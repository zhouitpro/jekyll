$c1 = array(20,5,2,99);     //顶级数组

$c2 = array(20,5);
$c3 = array(99,2,5);
$c4 = array(99,20);

//合并数组
$new_arr = array_merge($c1,$c3);

//移除重复的值
$new_arr = array_unique($new_arr);

var_dump($new_arr==$c1);
