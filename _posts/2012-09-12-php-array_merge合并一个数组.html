$array = array('key3'=>'key_string1', 'key4'=>'key_string2');

$array = array_merge($array, array('key1'=>'hello world'));

/**
 * array_merge 后面一个数组附加在前面一个数组的后面,如果存在相同的键,后面一个数组会覆盖前面一个数组
 */
var_dump($array);
