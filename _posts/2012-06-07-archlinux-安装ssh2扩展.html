#pacman -S libssh2 openssh openssl autoconf
下载http://aur.archlinux.org/packages/ph/php-ssh/php-ssh.tar.gz 
#wget http://aur.archlinux.org/packages/ph/php-ssh/php-ssh.tar.gz
#tar -xvf php-ssh.tar.gz
#cd php-ssh
#sudo makepkg --asroot php-ssh.install 
#pacman -U php-ssh-0.11.3-2-x86_64.pkg.tar.xz
为php手动添加扩展 
vim /etc/php/php.ini
添加 
extension=ssh2.so
重启apache 
sudo /etc/rc.d/httpd restart
