String InputValue = "";

do{
	
	Scanner in = new Scanner(System.in);

	System.out.println("请输入6位以上的字符串!");
	
	//获取输入值
	InputValue = in.nextLine();
	
	//循环判断输入的内容是否小于6位
}while(InputValue.length() < 6);

System.out.println("你输入的字符串是:"+InputValue);
