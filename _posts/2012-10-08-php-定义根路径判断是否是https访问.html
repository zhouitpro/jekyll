//定义完整的路径
define('BASEPATH',str_replace('\\','/',dirname(__FILE__)).'/');

//判断是否是https访问
$http = isset($_SERVER['HTTPS']) ? 'https://' : 'http://';

//定义http url
define('HTTPPATH',$http.$_SERVER['HTTP_HOST'].dirname($_SERVER['SCRIPT_NAME']));
