/**
* Implements template_preprocess_html().
*/
function YOUR_THEME_preprocess_html(&$vars) {
  $_meta = array(
    '#tag' => 'meta',
    '#attributes' => array(
      'name' => 'Creator',
      'content' => 'this is your custom content',
    ),
  );
  drupal_add_html_head($_meta, 'my_meta');
}
