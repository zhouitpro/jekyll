#/etc/init.d/mysql stop        //停止mysql服务
#mv /var/lib/mysql  /src/     //mysql默认目录存放在/var/lib/mysql
#vim /etc/mysql/my.cnf       //找到datadir 改成你现在的目录
#/etc/init.d/mysql restart   //重启mysql服务完成
