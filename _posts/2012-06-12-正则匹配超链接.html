$string = '<a href="http://www.ssh110.com/afds&aa">a</a><a href="http://drupal.ssh110.com/afds&aa">b</a>';
/**
 * 错误的 会进行贪婪匹配,也就是尽量多地去匹配
 * 之前使用此方法导致页面大乱
 */
$pattern = '#(href=["\'](.+)[\'"])#';

/*这样其实也行*/
$pattern = '#(href=["\']([^\'\"]+)[\'"])#';

//这样也其也行,因为加了?，使其变得不再贪婪
$pattern = '#(href=(["\']([\s\S]*?)["\'])#';
