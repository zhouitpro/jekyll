/**
 * 验证一个Function 是否存在
 */
function demo(){
    
}

$demo = 'demo';

var_dump(function_exists($demo));
var_dump(is_callable($demo));


/**
 * 验证一个类的方法是否存在
 */
Class Demo{
    
    function test(){
        
    }
}
$isc = array(new Demo(),'test');
var_dump(is_callable($isc));
var_dump(method_exists('Demo','test'));
var_dump(get_class_methods('Demo'));
