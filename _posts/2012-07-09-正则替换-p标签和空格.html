$string = 'nbsfdpfds;<p>&nbsp;fdskfjsdl&nbsp;fsdfsd</p>fsdfdfd';

if(preg_match('/<p>([\s\S]*?)<\/p>/', $string,$match)){
    if(isset($match[1])){
         $string = str_replace($match[0],  str_replace('&nbsp;', '', $match[1]), $string);
    }
}
