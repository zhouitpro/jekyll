pkgfile --update

vim ~/.bashrc
source /usr/share/doc/pkgfile/command-not-found.bash
登陆图像， 当你打开终端后会随机出现一些好看的图像: 
#pacman -S fortune-mod
#pacman -S cowsay

$vim ~/.bashrc

#放在第一行
fortune -a | fmt -80 -s | cowsay -$(shuf -n 1 -e b d g p s t w y) -f $(shuf -n 1 -e $(cowsay -l | tail -n +2)) -n
Bash 颜色, 放后面: https://wiki.archlinux.org/index.php/Color_Bash_Prompt 
vim ~/.bashrc

PS1="\[\033[01;37m\]\$? \$(if [[ \$? == 0 ]]; then echo \"\[\033[01;32m\];)\"; else echo \"\[\033[01;31m\];(\"; fi) $(if [[ ${EUID} == 0 ]]; then echo '\[\033[01;31m\]\h'; else echo '\[\033[01;32m\]\u@\h'; fi)\[\033[01;34m\] \w \$\[\033[00m\] "
安装pacman-color 
$sudo yaourt pacman-color
$vim .bashrc
 alias pacman='sudo pacman-color'
 alias grep='grep -n --color=auto'
