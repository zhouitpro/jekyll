RewriteEngine On

RewriteCond %{HTTP_HOST} !^(www.+|localhost|127\.0\.0\.1|.+?\..+?\..+)$ [NC]
RewriteRule ^(.*)$ http://www.%{HTTP_HOST}/$1 [R=301,L]

RewriteCond %{REQUEST_FILENAME} !-f 
RewriteCond %{REQUEST_FILENAME} !-d 
RewriteRule ^(.*)$ index.php/$1 [QSA,L]
