//声明一个数组
var array = new Array();	

//创建一个数组
var array = [];

//创建数组元素
array[0] = 'zhouitpro';
array[1] = '19';

//创建字符串下标
array['name'] = 'zhouitpro';
array['age'] = 19;
alert(array['name']);

alert(array.length);


/对数组添加元素
array.push('hello','world');
array.pop();  //移除数组最后一个元素
array.shift();  //移除数组开头的元素
array.unshift('yes');  //在数组前端添加一个元素

//排序
array.reverse();   //对数组进行逆向排序
array.sort();      //从小到大排序
alert(array);
