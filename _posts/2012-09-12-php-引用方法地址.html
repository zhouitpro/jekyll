function &static_save($name, $default_value = array())
{   
    static $data = array();
    
    if(isset($data[$name]))
        return $data[$name];
    
    $data[$name] = $default_value;
    
    return $data[$name];
}

$demo = &static_save('demo');

$demo['hello'] = 'hello';

$demo2 = &static_save('demo');

$demo2['haha'] = 'haha';

$demo3 = &static_save('demo');

var_dump($demo3);
输出结果: array 'hello' => string 'hello' (length=5) 'haha' => string 'haha' (length=4) 键为demo的值将永远保持不变
