获取周1和周末时间

/**
http://php.net/manual/en/function.strtotime.php
*/
$sunday = strtotime("sunday",time());
$monday = strtotime("-1 monday",time());

计算出今天的开始和结束的时间戳

$start_time = strtotime(date('Y-m-d'));
$end_time = strtotime('+1 day', $start_time);
